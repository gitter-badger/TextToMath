1 module Calculator.Parser(
    2     fixNegs,
    3     parse
    4 ) where
    5 
    6 import Calculator.Data.Token
    7 import Data.Maybe
    8 
    9 fixNegs :: [Token] -> [Token]
   10 fixNegs tokens = fixNegsAcc tokens [] Nothing
   11 
   12 fixNegsAcc :: [Token] -> [Token] -> Maybe Token -> [Token]
   13 fixNegsAcc [] acc _ = reverse acc
   14 fixNegsAcc (t:ts) acc prev
   15     | isNothing prev =
   16         if getKind t == Op && getLex t == "-" then
   17             fixNegsAcc ts (negToken:acc) $ Just negToken
   18         else
   19             fixNegsAcc ts (t:acc) $ Just t
   20     | otherwise =
   21         if (getKind t == Op && getLex t == "-") && (prevKind `elem` [Function, Op, Lparen]) then
   22               fixNegsAcc ts (negToken:acc) $ Just negToken
   23         else
   24             fixNegsAcc ts (t:acc) $ Just t
   25    where prevKind = getKind $ fromJust prev
   26          negToken = Token Function "neg"        
   27 
   28 parse :: [Token] -> [Token]
   29 parse tokens = parseAcc tokens [] []
   30 
   31 parseAcc :: [Token] -> [Token] -> [Token] -> [Token]
   32 parseAcc [] output [] = reverse output
   33 parseAcc [] output (op:ops)
   34     | getKind op == Lparen = error "ERROR: mismatched brackets"
   35     | otherwise = parseAcc [] (op:output) ops
   36 parseAcc tokens@(t:ts) output opStack
   37     | kind `elem` [Numeric, Id] = parseAcc ts (t:output) opStack
   38     | kind == Op =
   39         if null opStack || getKind (head opStack) == Lparen || precedence t > precedence (head opStack) then
   40             parseAcc ts output $ t:opStack
   41         else
   42             parseAcc tokens (head opStack : output) $ tail opStack
   43     | kind == Lparen = parseAcc ts output (t:opStack)
   44     | kind == Rparen =
   45         if null opStack then
   46             error "ERROR: mismatched brackets"
   47         else
   48             let op = head opStack
   49             in if getKind op == Lparen then
   50                 parseAcc ts output $ tail opStack
   51             else
   52                 parseAcc tokens (op:output) $ tail opStack
   53     where kind = getKind t
   54 
   55 precedence :: Token -> Int
   56 precedence t = case getLex t of
   57     "%" -> 0
   58     "+" -> 1
   59     "-" -> 1
   60     "*" -> 2
   61     "/" -> 2
   62     "^" -> 3
   63     _ -> 4 -- functions

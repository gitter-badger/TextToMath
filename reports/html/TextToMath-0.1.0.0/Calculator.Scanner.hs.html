1 module Calculator.Scanner (
    2     scan,
    3     Kind(..),
    4     Token(..)
    5 ) where
    6 
    7 import Data.Char(isAlpha,isDigit,isSpace)
    8 import Data.List
    9 import Data.Maybe
   10 import Control.Applicative()
   11 import Control.Exception()
   12 import Calculator.Data.Token
   13 
   14 data State = ST_START
   15            | ST_INT
   16            | ST_WS
   17            | ST_DEC
   18            | ST_E
   19            | ST_EXP
   20            | ST_OP
   21            | ST_ID
   22            | ST_LPAREN
   23            | ST_RPAREN
   24            | ST_NULL
   25            deriving(Eq,Show)
   26 
   27 data Transition = Transition {
   28     getState :: State,
   29     getCharSet :: Char -> Bool,
   30     getToState :: State
   31 }
   32 
   33 scan :: String -> [Token]
   34 scan str = scanAcc str transitionTable ST_START finalStates "" []
   35 
   36 scanAcc :: String -> [Transition] -> State -> [State] -> String -> [Token] -> [Token]
   37 scanAcc [] _ cState fStates lexAcc tokAcc
   38     | cState `elem` fStates =
   39         reverse $
   40         if cState == ST_WS then
   41             tokAcc
   42         else
   43             makeToken cState (reverse lexAcc) : tokAcc
   44     | otherwise = error "ERROR: unexpected end of string"
   45 
   46 scanAcc str@(chr:chrs) trTable cState fStates lexAcc tokAcc =
   47     let
   48         foundTrans state c t = state == getState t && getCharSet t c
   49         tr = find (foundTrans cState chr) trTable
   50     in if isNothing tr then
   51         if cState `elem` fStates then
   52             scanAcc str trTable ST_START fStates [] (
   53                 if cState == ST_WS then
   54                     tokAcc
   55                 else
   56                     makeToken cState (reverse lexAcc) : tokAcc
   57             )
   58         else
   59             error $ "ERROR: Left to parse " ++ show cState ++ " " ++ str
   60     else
   61         let
   62             toState = getToState $ fromJust tr
   63         in if toState == ST_NULL then
   64             error $ "ERROR: Invalid symbol " ++ [chr]
   65         else
   66             scanAcc chrs trTable toState fStates (chr:lexAcc) tokAcc
   67 
   68 makeToken :: State -> String -> Token
   69 makeToken state lexeme
   70     | kind == Numeric && head lexeme == '.' = Token kind ('0':lexeme)
   71     | kind == Id =
   72         if lexeme `elem` ["sin","cos","tan","asin","acos","atan"] then
   73            Token Function lexeme
   74         else
   75             Token Id lexeme
   76     | otherwise = Token kind lexeme
   77     where kind = stateToKind state
   78 
   79 stateToKind :: State -> Kind
   80 stateToKind state
   81     | state `elem` [ST_INT, ST_DEC, ST_EXP] = Numeric
   82     | state == ST_OP = Op
   83     | state == ST_ID = Id
   84     | state == ST_LPAREN = Lparen
   85     | state == ST_RPAREN = Rparen
   86 
   87 isOperator :: Char -> Bool
   88 isOperator = flip elem "+-/*^%"
   89 
   90 notMinus :: Char -> Bool
   91 notMinus x = isOperator x && x /= '-'
   92 
   93 notE :: Char -> Bool
   94 notE x = isAlpha x && x /= 'e'
   95 
   96 transitionTable :: [Transition]
   97 transitionTable = 
   98     [ Transition ST_START isDigit ST_INT
   99     , Transition ST_START isAlpha ST_ID
  100     , Transition ST_START isSpace ST_WS
  101     , Transition ST_START isOperator ST_OP
  102     , Transition ST_START (== '(') ST_LPAREN
  103     , Transition ST_START (== ')') ST_RPAREN
  104     , Transition ST_START (== '.') ST_DEC
  105     , Transition ST_INT isDigit ST_INT
  106     , Transition ST_INT (== '.') ST_DEC
  107     , Transition ST_INT (== 'e') ST_E
  108     , Transition ST_E (\x -> x == '-' || isDigit x) ST_EXP
  109     , Transition ST_INT notE ST_NULL
  110     , Transition ST_DEC isDigit ST_DEC
  111     , Transition ST_DEC (== '.') ST_NULL
  112     , Transition ST_DEC (== 'e') ST_EXP
  113     , Transition ST_DEC notE ST_NULL
  114     , Transition ST_EXP isDigit ST_EXP
  115     , Transition ST_EXP (== '.') ST_NULL
  116     , Transition ST_EXP isAlpha ST_NULL
  117     , Transition ST_ID isAlpha ST_ID
  118     , Transition ST_ID isDigit ST_NULL
  119     , Transition ST_ID (== '.') ST_NULL
  120     , Transition ST_OP notMinus ST_NULL
  121     ]
  122 
  123 finalStates :: [State]
  124 finalStates =
  125     [ ST_INT
  126     , ST_WS
  127     , ST_DEC
  128     , ST_EXP
  129     , ST_OP
  130     , ST_ID
  131     , ST_LPAREN
  132     , ST_RPAREN
  133     ]
